{KDJ_B1}
短线周期:=28;
中线周期:=57;
长线周期:=114;
RSV:=(CLOSE-LLV(LOW,9))/(HHV(HIGH,9)-LLV(LOW,9))*100;
K:SMA(RSV,3,1);
D:SMA(K,3,1);
J:3*K-2*D;
{J值有拐点、J小于55、J在D、K线下方}
J拐点:=J<REF(J,1) AND J<REFX(J,1) AND J<55 AND J<D AND J<K AND K<D;
MIN_J短:=SUM(IF(J拐点=1,J,0),短线周期)/MAX(SUM(IF(J拐点=1,1,0),短线周期),1);
MIN_J中:=SUM(IF(J拐点=1,J,0),中线周期)/MAX(SUM(IF(J拐点=1,1,0),中线周期),1);
MIN_J长:=SUM(IF(J拐点=1,J,0),长线周期)/MAX(SUM(IF(J拐点=1,1,0),长线周期),1)+10;
MIN_J:(MIN_J短+MIN_J中+MIN_J长)/3.0;
